# ТЕХНІЧНЕ ЗАВДАННЯ
# Проект: Portable Indoor Feedback (PIF) — Вебсайт
# Версія ТЗ: 2.0
# Стек: PHP 8.x, jQuery 3.x, MariaDB, Bootstrap 5, Chart.js, WebSocket (Ratchet)

---

## 1. ЗАГАЛЬНИЙ ОПИС

Вебплатформа для моніторингу мікроклімату в приміщеннях. Станції (Raspberry Pi + сенсори BME280, BH1750, MQ-135) надсилають виміри кожні ~2 секунди на сервер. Користувачі через вебсайт переглядають дані в реальному часі, створюють колекції, діляться ними з друзями та спілкуються у вбудованому чаті.

## 4. СХЕМА БАЗИ ДАНИХ (MariaDB)

```sql
-- Користувачі
CREATE TABLE user (
    pk_username    VARCHAR(50)   NOT NULL PRIMARY KEY,
    first_name     VARCHAR(50)   NOT NULL,
    last_name      VARCHAR(50)   NOT NULL,
    email          VARCHAR(100)  NOT NULL UNIQUE,
    password_hash  VARCHAR(255)  NOT NULL,          -- bcrypt
    is_admin       TINYINT(1)    NOT NULL DEFAULT 0,
    avatar_path    VARCHAR(255)  DEFAULT NULL,       -- шлях до аватарки в /uploads/avatars/
    language       ENUM('en','fr','uk') NOT NULL DEFAULT 'en',
    theme          ENUM('light','dark') NOT NULL DEFAULT 'light',
    created_at     DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Станції
CREATE TABLE station (
    pk_serial      VARCHAR(20)   NOT NULL PRIMARY KEY,  -- напр. SN-1001
    name           VARCHAR(100)  DEFAULT NULL,
    description    VARCHAR(1000) DEFAULT NULL,
    fk_owner       VARCHAR(50)   DEFAULT NULL,          -- NULL = ще не зареєстрована
    FOREIGN KEY (fk_owner) REFERENCES user(pk_username) ON DELETE SET NULL
);

-- Виміри (вставляються кожні ~2 секунди)
CREATE TABLE measurement (
    pk_id          INT(11)       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    fk_station     VARCHAR(20)   NOT NULL,
    timestamp      DATETIME(3)   NOT NULL,             -- мілісекунди для точності
    temperature    FLOAT         NOT NULL,              -- °C
    humidity       FLOAT         NOT NULL,              -- %
    pressure       FLOAT         NOT NULL,              -- hPa
    light          FLOAT         NOT NULL,              -- lux
    gas            FLOAT         NOT NULL,              -- ppm
    INDEX idx_station_time (fk_station, timestamp),    -- для швидкої фільтрації
    FOREIGN KEY (fk_station) REFERENCES station(pk_serial) ON DELETE CASCADE
);

-- Колекції (збірки вимірів)
CREATE TABLE collection (
    pk_id          INT(11)       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name           VARCHAR(100)  NOT NULL,
    description    VARCHAR(1000) DEFAULT NULL,
    fk_owner       VARCHAR(50)   NOT NULL,
    created_at     DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (fk_owner) REFERENCES user(pk_username) ON DELETE CASCADE
);

-- Зв'язок колекція ↔ вимір (M:M)
-- Виміри додаються явно (вручну або через діапазон дат при створенні/додаванні)
-- Якщо при додаванні вказано end_dt в майбутньому — додати всі існуючі виміри,
-- а нові що з'являться — додавати автоматично через тригер або cron/WS-сервер
CREATE TABLE collection_measurement (
    fk_collection  INT(11)       NOT NULL,
    fk_measurement INT(11)       NOT NULL,
    PRIMARY KEY (fk_collection, fk_measurement),
    FOREIGN KEY (fk_collection)  REFERENCES collection(pk_id)   ON DELETE CASCADE,
    FOREIGN KEY (fk_measurement) REFERENCES measurement(pk_id)  ON DELETE CASCADE
);

-- Активні "підписки" колекцій на майбутні виміри
-- Якщо end_dt > NOW() при додаванні — створюється запис тут
-- WS-сервер після кожного ingest перевіряє цю таблицю і додає новий вимір в потрібні колекції
CREATE TABLE collection_subscription (
    pk_id          INT(11)       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    fk_collection  INT(11)       NOT NULL,
    fk_station     VARCHAR(20)   NOT NULL,
    end_dt         DATETIME      NOT NULL,             -- коли зупинити автододавання
    FOREIGN KEY (fk_collection) REFERENCES collection(pk_id) ON DELETE CASCADE,
    FOREIGN KEY (fk_station)    REFERENCES station(pk_serial) ON DELETE CASCADE
);

-- Поділення колекцій
CREATE TABLE collection_share (
    fk_collection  INT(11)       NOT NULL,
    fk_user        VARCHAR(50)   NOT NULL,
    PRIMARY KEY (fk_collection, fk_user),
    FOREIGN KEY (fk_collection) REFERENCES collection(pk_id)      ON DELETE CASCADE,
    FOREIGN KEY (fk_user)       REFERENCES user(pk_username)       ON DELETE CASCADE
);

-- Дружба (з запитами)
CREATE TABLE friendship (
    pk_id          INT(11)       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    fk_sender      VARCHAR(50)   NOT NULL,
    fk_receiver    VARCHAR(50)   NOT NULL,
    status         ENUM('pending','accepted','declined') NOT NULL DEFAULT 'pending',
    created_at     DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_pair (fk_sender, fk_receiver),
    FOREIGN KEY (fk_sender)   REFERENCES user(pk_username) ON DELETE CASCADE,
    FOREIGN KEY (fk_receiver) REFERENCES user(pk_username) ON DELETE CASCADE
);

-- Повідомлення чату
CREATE TABLE chat_message (
    pk_id          INT(11)       NOT NULL AUTO_INCREMENT PRIMARY KEY,
    fk_sender      VARCHAR(50)   NOT NULL,
    fk_receiver    VARCHAR(50)   NOT NULL,
    message        TEXT          DEFAULT NULL,         -- NULL якщо це файл
    file_path      VARCHAR(255)  DEFAULT NULL,         -- шлях до файлу в /uploads/chat/
    file_name      VARCHAR(255)  DEFAULT NULL,         -- оригінальна назва файлу
    file_type      VARCHAR(100)  DEFAULT NULL,         -- MIME тип
    sent_at        DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    is_read        TINYINT(1)    NOT NULL DEFAULT 0,
    FOREIGN KEY (fk_sender)   REFERENCES user(pk_username) ON DELETE CASCADE,
    FOREIGN KEY (fk_receiver) REFERENCES user(pk_username) ON DELETE CASCADE
);
```

## 5. ЛОГІКА КОЛЕКЦІЙ (ВАЖЛИВО!)

Колекція — це **іменований список конкретних вимірів** (зв'язок через `collection_measurement`).

### При створенні / додаванні вимірів до колекції:
Користувач вибирає **станцію + start_dt + end_dt** як зручний спосіб вибірки.
API виконує:
```sql
-- 1. Знайти всі виміри за цим діапазоном
SELECT pk_id FROM measurement
WHERE fk_station = :station AND timestamp BETWEEN :start AND :end;

-- 2. Вставити їх у collection_measurement (INSERT IGNORE для дублів)
INSERT IGNORE INTO collection_measurement (fk_collection, fk_measurement) VALUES ...

-- 3. Якщо end_dt > NOW() — створити підписку
INSERT INTO collection_subscription (fk_collection, fk_station, end_dt) VALUES ...
```

### Автододавання майбутніх вимірів:
WS-сервер після кожного успішного `ingest` перевіряє `collection_subscription`:
```sql
SELECT fk_collection FROM collection_subscription
WHERE fk_station = :station AND end_dt >= NOW();
-- → INSERT IGNORE INTO collection_measurement для нового measurement pk_id
-- → DELETE підписки де end_dt < NOW() (прибирати протерміновані)
```

### Виміри можна додавати з різних станцій:
Немає обмеження — один виклик `add_measurements.php` вказує одну станцію+діапазон,
але можна викликати декілька разів з різними станціями для однієї колекції.

## 6. REAL-TIME ЧЕРЕЗ WEBSOCKET

WebSocket сервер (порт 8080) обробляє **три типи повідомлень**:

### 6.1 Авторизація при підключенні
```json
→ { "type": "auth", "session_id": "abc123" }
← { "type": "auth_ok", "username": "user1" }
```
Сервер читає файл PHP сесії з session_id, дістає `$_SESSION['username']`.

### 6.2 Чат
```json
→ { "type": "chat", "to": "user2", "text": "Привіт!" }
← { "type": "chat", "from": "user1", "text": "Привіт!", "sent_at": "2026-02-21T12:00:00" }
```
WS-сервер зберігає повідомлення в `chat_message` і пушить отримувачу якщо онлайн.

### 6.3 Push нових вимірів (subscribe)
```json
→ { "type": "subscribe_station", "station": "SN-1001" }
← { "type": "measurement", "station": "SN-1001", "data": { "temperature":22.5, ... } }
```
Клієнт підписується на станцію. WS-сервер після кожного ingest пушить нові дані
всім підписникам цієї станції. Dashboard та measurements.php використовують це для
реального часу без polling.

### 6.4 Push файлів у чаті
```json
← { "type": "chat_file", "from": "user1", "file_url": "/uploads/chat/abc.jpg",
     "file_name": "photo.jpg", "file_type": "image/jpeg", "sent_at": "..." }
```

## 8. ФРОНТЕНД — СТОРІНКИ (PHP файли з include)

Кожна сторінка в public/ — це .php файл, який:
1. `require_once 'includes/auth_check.php';` — редірект якщо не залогінений
2. `require_once 'includes/head.php';` — Bootstrap, jQuery, Chart.js, i18n, theme

## 11. ЦЕНТРАЛІЗОВАНИЙ API МОДУЛЬ (api.js)

```javascript
// assets/js/api.js
const API = {
    base: '/api',

    _req(method, endpoint, data, isForm) {
        const opts = {
            url: this.base + endpoint,
            method,
            error(xhr) {
                if (xhr.status === 401) location.href = '/index.php';
                else console.error(xhr.responseJSON?.error || xhr.statusText);
            }
        };
        if (isForm) {
            opts.data = data;           // FormData для file upload
            opts.processData = false;
            opts.contentType = false;
        } else if (data) {
            opts.contentType = 'application/json';
            opts.data = JSON.stringify(data);
        }
        return $.ajax(opts);
    },

    get:    (url, params) => API._req('GET', url + (params ? '?' + $.param(params) : '')),
    post:   (url, data)   => API._req('POST', url, data),
    put:    (url, data)   => API._req('PUT', url, data),
    del:    (url, data)   => API._req('DELETE', url, data),
    upload: (url, form)   => API._req('POST', url, form, true),
};
```

## 12. WEBSOCKET МЕНЕДЖЕР (ws.js)

```javascript
// assets/js/ws.js
const WS = {
    conn: null,
    handlers: {},
    reconnectDelay: 2000,

    connect() {
        this.conn = new WebSocket('ws://' + location.hostname + ':8080');
        this.conn.onopen  = () => { this.send({ type: 'auth', session_id: SESSION_ID }); };
        this.conn.onclose = () => { setTimeout(() => this.connect(), this.reconnectDelay); };
        this.conn.onmessage = (e) => {
            const msg = JSON.parse(e.data);
            (this.handlers[msg.type] || []).forEach(fn => fn(msg));
        };
    },

    on(type, fn)   { (this.handlers[type] ??= []).push(fn); },
    send(data)     { this.conn?.send(JSON.stringify(data)); },
    subscribe(sn)  { this.send({ type: 'subscribe_station', station: sn }); },
    unsubscribe(sn){ this.send({ type: 'unsubscribe_station', station: sn }); },
};

// SESSION_ID — вставляється з PHP в head.php:
// <script>const SESSION_ID = '<?= session_id() ?>';</script>
```

## 17. MAILHOG (ПОШТА)

MailHog — це SMTP сервер-пастка для розробки. Він приймає всі листи але **не відправляє** їх реально. Листи можна переглянути на http://localhost:8025.

### PHPMailer конфіг (api/config/mail.php)
```php
<?php
use PHPMailer\PHPMailer\PHPMailer;

function sendMail(string $to, string $subject, string $body): void {
    $mail = new PHPMailer(true);
    $mail->isSMTP();
    $mail->Host       = 'mailhog';   // ім'я Docker сервісу
    $mail->Port       = 1025;        // MailHog SMTP порт
    $mail->SMTPAuth   = false;       // MailHog не потребує авторизації
    $mail->setFrom('noreply@pif.local', 'Portable Indoor Feedback');
    $mail->addAddress($to);
    $mail->isHTML(true);
    $mail->Subject = $subject;
    $mail->Body    = $body;
    $mail->send();
}
```

### Встановити PHPMailer
```bash
# В корені проекту
composer require phpmailer/phpmailer
```

### Коли надсилаються листи
- Реєстрація → welcome email
- Запит на дружбу → email отримувачу
- Прийнятий запит → email відправнику
- Поділена колекція → email отримувачу

## 18. ВИМОГИ ДО КОДУ

1. **Структура**: API тільки в `api/`, фронтенд в `public/`
2. **PHP у фронтенді**: дозволено для `include`, перевірки сесії, виведення SESSION_ID — але без SQL запитів
3. **jQuery**: весь DOM + AJAX через jQuery. WebSocket через нативний JS (WS API не підтримує jQuery)
4. **Без дублювань**: `api.js` для всіх запитів, `db.php` для БД, `ws.js` для WS
5. **Коментарі**: кожна функція прокоментована (PHPDoc для PHP, JSDoc для JS)
6. **Безпека**: PDO prepared statements скрізь, bcrypt (`password_hash`), `session_regenerate_id()` при логіні, CORS обмежений
7. **Валідація**: jQuery на клієнті + PHP на сервері
8. **Responsive**: Bootstrap 5 grid, перевірити на мобільному, планшеті, десктопі
9. **Підтримка тем**: Bootstrap `data-bs-theme`, мінімум кастомного CSS
